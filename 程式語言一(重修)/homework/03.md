#include<stdio.h> 

int main(){
	
	int i;
	int count=0;
	int avg=0;
	
	printf("計算平均分數:\n-------------------------------------\n");
	
	
	printf("輸入人數: ");
	scanf("%d",&count);
	
	int num[count];
	for(i=0;i<count;i++){
		
		printf("學生第%d個 分數為: ",i+1);
		scanf("%d",&num[i]);
		
		avg+=num[i];	
	}
	avg=avg/count;
	printf("平均分數:%d",avg);
	
	return 0;
}
