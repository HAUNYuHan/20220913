#include <stdio.h>

int main(){
	
	int i =0;
	int number[5]={0,1,2,3,4,};
	int length =sizeof(number)/sizeof(number[0]);
	int sum;
	
	
	printf("sizeof(number)=%d sizeof(number[0])=%d\n",sizeof(number),sizeof(number[0]));
	printf("length=%d\n",length);
	
	for(i;i<length;i++){
		if(i!=0){
			number[i]=90;
		}
		printf("%d ",number[i]);
		sum=sum+number[i];
	}
	printf("\n%d",sum/length);
	printf("\n");
	return 0;
}

#include<stdio.h> 
#define ROW 5
#define COLUMN 10
int main(){
	int maze[ROW][COLUMN];
	int i=0,j=0;
	for(i=0;i<ROW;i++){
		for(j=0;j<COLUMN;j++){
			maze[i][j]=(i+j)*(j+1);
		}
	}
	for(i=0;i<ROW;i++){
		for(j=0;j<COLUMN;j++){
			printf("%d\t",maze[i][j]);
		}
		putchar('\n');
	}
	
	return 0;
}
