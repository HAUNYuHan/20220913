#include <stdio.h>

int main( ){
	int score = 0;
	
	printf("輸入分數:");
	scanf("%d",&score);
	if(score >= 90){
		puts("得A");
	}else if(score >= 80 && score <90){
		puts("得B");
	}else if(score >= 70 && score <80){
		puts("得C");
	}
	else if(score >= 60 && score <70){
		puts("得D");
	}else{
		puts("得E(不及格)");
	}
	return 0;
}
