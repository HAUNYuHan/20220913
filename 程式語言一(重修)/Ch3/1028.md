	int score = 0;
	
	printf("輸入分數:");

	if(score >= 90){
		puts("得A");
	}else if(score >= 80 && score <90){
		puts("得B");
	}else if(score >= 70 && score <80){
		puts("得C");
	}
	else if(score >= 60 && score <70){
		puts("得D");
	}else{
		puts("得E(不及格)");
	}
	return 0;
}

## 九九乘法表
```
#include <stdio.h>

int main( ){
	for(int j = 1; j < 10; j++){
		for(int i = 2; i<10; i++){
			printf("%d*%d=%2d",i,j,i*j);
		}
		puts("");
	} 
	return 0;
}
```
