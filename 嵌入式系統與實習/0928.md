const int ledPin[]={3,9,10,11};
int led_length = sizeof(ledPin)/sizeof(ledPin[0]);
int i;
int j=0;
int direct=0;
void setup(){
  for(i=0;i<led_length;i++)
    pinMode(ledPin[i],OUTPUT);
}
void loop(){
  for(i=0;i<led_length;i++){
    digitalWrite(ledPin[i],LOW);
  }
  digitalWrite(ledPin[j],HIGH);
  delay(100);

  if(direct==0){
    if(j==led_length-1)
      direct=1;
    else
        j++;
  }else{
    if(j==0)
      direct=0;
    else
        j--;
  }
}
