```
#include<Servo.h>

Servo servo;
int angle;
int shaft_angle = 15;
const int servoPin = 2;

void setup(){
  const int msmin = 600;
  const int msmax = 2400;
  servo.attach(servoPin, msmin, msmax );
}

void loop(){
  for(angle=0;angle<=180;angle=angle+shaft_angle){
    servo.write(angle);
    delay(500);
  }

  for(angle=180; angle>=0;angle-shaft_angle){
    servo.write(angle);
    delay(100);
  }
}
```
